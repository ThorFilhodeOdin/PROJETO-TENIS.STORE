

                --REGRAS DO BC--


Tabela: Categorias
Essa tabela vai armazenar as categorias de tênis disponíveis na loja (ex: esportivo, casual, feminino, masculino, etc).
CREATE TABLE Categorias (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nome_categoria VARCHAR(255) NOT NULL
);


Tabela: Produtos
Aqui, você terá as informações sobre os tênis, como o nome, preço, quantidade, categoria (relacionada à tabela Categorias), e outros detalhes.
CREATE TABLE Produtos (
    id_produto INT AUTO_INCREMENT PRIMARY KEY,
    nome_produto VARCHAR(255) NOT NULL,
    descricao TEXT,
    preco DECIMAL(10, 2) NOT NULL,
    quantidade_estoque INT NOT NULL,
    id_categoria INT,
    FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria)
);


Tabela: Clientes
Essa tabela vai armazenar informações sobre os clientes da loja.
CREATE TABLE Clientes (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    cpf INT NOT NULL,
    nome_cliente VARCHAR(255) NOT NULL,
    email_cliente VARCHAR(255) UNIQUE NOT NULL,
    telefone_cliente VARCHAR(15),
    endereco_cliente TEXT
);


Tabela: Vendas
Registra as vendas realizadas na loja. Cada venda pode ter vários itens (produtos).
CREATE TABLE Vendas (
    id_venda INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    data_venda DATETIME NOT NULL,
    total_venda DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (id_cliente) REFERENCES Clientes(id_cliente)
);


Tabela: Itens_Venda
Essa tabela vai associar os produtos vendidos a uma venda específica, com a quantidade e o preço de cada item.
CREATE TABLE Itens_Venda (
    id_item_venda INT AUTO_INCREMENT PRIMARY KEY,
    id_venda INT,
    id_produto INT,
    quantidade INT NOT NULL,
    preco DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (id_venda) REFERENCES Vendas(id_venda),
    FOREIGN KEY (id_produto) REFERENCES Produtos(id_produto)
);
